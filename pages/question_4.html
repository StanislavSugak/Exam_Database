DECLARE @MyVariable VARCHAR(50) //создание переменных для курсора

DECLARE MyCursor CURSOR FOR //создание курсора

SELECT MyColumn

FROM MyTable

OPEN MyCursor //открытие курсора

FETCH NEXT FROM MyCursor INTO @MyVariable //считывание данных из курсора в переменную

- PRIOR // предыдущая

- FIRST // на первую

- LAST // на последнюю

- ABSOLUTE n //строка, которая является n-ой в курсоре

- RELATIVE n //строка, которая строит на n от текущей

WHILE @@FETCH_STATUS = 0 //все успешно, курсор переместился на новую строку

1 //последняя операция не удалась, так как нету строк для извлечения

-1 //произошла ошибка при выполнении FETCH

-2 //курсор был закрыт или освобожден

BEGIN

PRINT @MyVariable

FETCH NEXT FROM MyCursor INTO @MyVariable

END

CLOSE MyCursor //закрытие курсора

DEALLOCATE MyCursor //освобождение памяти
